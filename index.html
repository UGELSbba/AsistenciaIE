Manual de Conexión a Google Sheets

Instrucciones para tener tu Base de Datos en Google Sheets:

  1. Abre un nuevo documento en Google Sheets.
  2. Ve al menú Extensiones > Apps Script.
  3. Borra el código que aparezca y pega el siguiente script de procesamiento:
function doPost(e) {
  var data = JSON.parse(e.postData.contents);
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  
  if (data.action === "save") {
    sheet.clearContents(); // Limpia y sobrescribe con la nueva información
    // Agregar cabecera
    sheet.appendRow(["DNI", "Nombre", "Cargo", "Condicion", "Jornada", "AsistenciaJSON"]);
    
    data.staff.forEach(function(member) {
      sheet.appendRow([
        member.dni, 
        member.name, 
        member.cargo, 
        member.condicion, 
        member.jornada, 
        JSON.stringify(member.asistencias)
      ]);
    });
    
    return ContentService.createTextOutput(JSON.stringify({status: "success"}))
      .setMimeType(ContentService.MimeType.JSON);
  }
  
  if (data.action === "load") {
    var rows = sheet.getDataRange().getValues();
    var staff = [];
    for (var i = 1; i < rows.length; i++) {
      staff.push({
        dni: rows[i][0],
        name: rows[i][1],
        cargo: rows[i][2],
        condicion: rows[i][3],
        jornada: parseInt(rows[i][4]),
        asistencias: JSON.parse(rows[i][5] || "{}")
      });
    }
    return ContentService.createTextOutput(JSON.stringify({status: "success", staff: staff}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
